CREATE TYPE listing_type_enum AS ENUM ('donation', 'sale');
CREATE TYPE book_condition_enum AS ENUM ('new', 'good', 'worn'); -- Yeni, İyi, Yıpranmış
CREATE TYPE listing_status_enum AS ENUM ('available', 'reserved', 'completed', 'inactive');

CREATE TABLE listings (
    id SERIAL PRIMARY KEY,
    user_id INT REFERENCES users(id) ON DELETE CASCADE,
    book_id INT REFERENCES books(id) ON DELETE RESTRICT,
    listing_type listing_type_enum NOT NULL,
    price NUMERIC(10, 2) DEFAULT 0.00,  -- Eğer 'donation' ise 0 olacak
    condition book_condition_enum NOT NULL,
    description TEXT,
    status listing_status_enum DEFAULT 'available',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
