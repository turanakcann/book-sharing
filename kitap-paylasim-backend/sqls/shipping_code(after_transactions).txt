CREATE TYPE shipping_provider_enum AS ENUM ('ptt', 'yurtici', 'mng', 'aras');
CREATE TYPE transaction_status_enum AS ENUM (
    'pending_payment',  -- Alıcı parayı/güvence bedelini ödedi (varsa), satıcının kargolaması bekleniyor
    'shipping_code_generated', -- Dolap'taki gibi otomatik kargo kodu üretildi
    'in_transit',       -- Satıcı kargoya verdi, yolda
    'delivered',        -- Kargo çalışanı teslim etti (24 saatlik sayaç başladı)
    'completed',        -- Alıcı onayladı veya 24 saat doldu, işlem bitti
    'cancelled'         -- İptal edildi
);

ALTER TABLE transactions ADD COLUMN shipping_provider shipping_provider_enum;
ALTER TABLE transactions ADD COLUMN shipping_code VARCHAR(50) UNIQUE; -- Satıcının kargoya vereceği kod
ALTER TABLE transactions ADD COLUMN shipped_at TIMESTAMP; -- Kargoya verilme tarihi
ALTER TABLE transactions ADD COLUMN delivered_at TIMESTAMP; -- Kargonun ulaştığı tarih (Sayaç başlangıcı)
ALTER TABLE transactions ADD COLUMN buyer_confirmed_at TIMESTAMP; -- Alıcının butona bastığı tarih
